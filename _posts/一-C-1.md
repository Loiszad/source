### 3.继承
```
class Person{
	private:
		int age;
	public:
		Person(int age);
		int getAge();	
		void sayhello();
}


Person::Person(int age){
	this->age = age;
}

int Person::getAge(){
	return this->age;
}

void Person::sayhello(){
	printf("hello");
}

//继承 public表示冲Person继承过来的成员是公开的
class Man: public Person{
	public:
		Man(int age);
}

//调用父类的构造方法
Man::Man(int age):Person(age){
	Person::sayhello();//调用父类的方法 
}


int main(){
	Man *m = new Man();
	m->Person::sayhello();//调用父类的方法
	
	return 0;
}
``` 
 
### 4.虚函数、纯虚函数、函数重写

```
class Person{
	private:
		int age;
	public:
		Person(int age);
		int getAge();	
		virtual void sayhello() = 0; //②纯虚函数，类似java的抽象函数，需要子类去实现
}
 
 
class Man: public Person{
	public:
		virtual void sayhello(){
			//实现父类的纯虚函数
		}
	 	
}
 
int main(){
	Person *p = new Man();
	p->sayhello(); // ①调用的是父类的方法  要想跟java一样动态的调用，需要定义成虚函数 virtual  
	
	return 0;
}
```
